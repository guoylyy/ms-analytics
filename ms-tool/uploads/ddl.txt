
     CREATE TABLE TAICANG.TAX_PREDICT(
          START_DATETIME TIMESTAMP,
          END_DATETIME TIMESTAMP,
          LAST_UPDATE_DATETIME TIMESTAMP,
          LOCATION VARCHAR(800),
         TIMEZONE_OFFSET  INTEGER,
          YEAR_MONTH INTEGER NOT NULL,
          SOURCE VARCHAR(50) NOT NULL,
          ZONE_NAME VARCHAR(50) NOT NULL,
          ACTUAL_AMOUNT DECIMAL(15,2) NOT NULL,
          PREDICT_AMOUNT DECIMAL(15,2),
          ACTUAL_GROWTH_RATE DECIMAL(15,2) NOT NULL,
          PREDICT_GROWTH_RATE DECIMAL(10,2),
          PRIMARY KEY(YEAR_MONTH,SOURCE, ZONE_NAME)
     )  ;


    CREATE TABLE TAICANG.TAX_FACT(
          YEAR_MONTH INTEGER NOT NULL,
          ZONE_NAME VARCHAR(50) NOT NULL,
          SOURCE VARCHAR(50) NOT NULL,
          TAX_AMOUNT  DECIMAL(15, 2) NOT NULL,
          YEAR CHAR(2),
          MONTH CHAR(4),
          PRIMARY KEY(YEAR_MONTH,ZONE_NAME,SOURCE) 
     ) ;
